  <script>
    const casti1 = ['mega', 'ultra', 'kyber', 'astro', 'bio', 'koko', 'neuro', 'techno'];
    const casti2 = ['kokot', 'čurák', 'píčus', 'debil', 'šulin', 'blboun', 'šmejd', 'trubka'];

    let aktualniSlovo = '';

    function vygeneruj() {
      const nahodna1 = casti1[Math.floor(Math.random() * casti1.length)];
      const nahodna2 = casti2[Math.floor(Math.random() * casti2.length)];
      aktualniSlovo = nahodna1 + '-' + nahodna2;
      document.getElementById('vystup').textContent = aktualniSlovo;
    }

    function ulozOblibene() {
      if (aktualniSlovo === '') return;
      const seznam = document.getElementById('oblibeneList');
      const polozka = document.createElement('li');
      polozka.textContent = aktualniSlovo;
      seznam.appendChild(polozka);
    }
